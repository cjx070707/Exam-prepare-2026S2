### Activation Functions（Forward + Backward）

```python
# Sigmoid
def forward(self, z):
    self.s = 1 / (1 + torch.exp(-z))
    return self.s

def backward(self, grad):
    return grad * self.s * (1 - self.s)

# ──────────────────────────────────────
# Tanh
def forward(self, z):
    self.out = torch.tanh(z)
    return self.out

def backward(self, grad):
    return grad * (1 - self.out ** 2)

# ──────────────────────────────────────
# ReLU
def forward(self, z):
    self.z = z
    return torch.clamp(z, min=0)

def backward(self, grad):
    return grad * (self.z > 0).float()
```

### HiddenLayer（Forward + Backward）

```python
def forward(self, X):
    self.X = X                           # (N, F_in)
    z = X @ self.W + self.b              # (N, F_out)
    return self.act.forward(z)           # activation applied

def backward(self, grad):
    dz = self.act.backward(grad)         # (N, F_out)
    self.dW = self.X.T @ dz             # (F_in, F_out)
    self.db = dz.sum(dim=0, keepdim=True)  # (1, F_out)
    dx = dz @ self.W.T                  # (N, F_in)
    return dx
```

### CrossEntropyLoss（Forward + Backward）

```python
def forward(self, logits, labels):
    # logits: (N, C), labels: (N,) int
    N = logits.shape[0]
    shifted = logits - logits.max(dim=1, keepdim=True).values  # numerical stability
    log_sum = torch.log(torch.exp(shifted).sum(dim=1, keepdim=True))
    log_probs = shifted - log_sum
    self.probs  = torch.exp(log_probs)  # save softmax probs
    self.labels = labels
    return -log_probs[torch.arange(N), labels].mean()

def backward(self):
    N = self.probs.shape[0]
    grad = self.probs.clone()            # (N, C)
    grad[torch.arange(N), self.labels] -= 1.0
    return grad / N
```

### Batch Normalization（Forward + Backward）

```python
# Forward
def forward(self, z):
    B = z.shape[0]
    mu    = z.mean(dim=0, keepdim=True)
    var   = z.var(dim=0, keepdim=True, unbiased=False)
    std   = torch.sqrt(var + 1e-5)
    z_hat = (z - mu) / std
    self._cache = (z_hat, std, B)
    return self.gamma * z_hat + self.beta

# Backward
def backward(self, grad):
    z_hat, std, B = self._cache
    self.dgamma = (grad * z_hat).sum(dim=0, keepdim=True)
    self.dbeta  =  grad.sum(dim=0, keepdim=True)
    dz_hat = grad * self.gamma
    dz = (1.0 / (B * std)) * (
        B * dz_hat
        - dz_hat.sum(dim=0, keepdim=True)
        - z_hat * (dz_hat * z_hat).sum(dim=0, keepdim=True)
    )
    return dz
```

### naive_conv2d（单样本，单通道理解版）

```python
def naive_conv2d(x, w, stride=1, padding=0):
    # x: (N, C_in, H, W)   w: (C_out, C_in, kH, kW)
    N, C_in, H, W = x.shape
    C_out, _, kH, kW = w.shape
    H_out = (H + 2*padding - kH) // stride + 1
    W_out = (W + 2*padding - kW) // stride + 1
    if padding > 0:
        x = F.pad(x, [padding]*4)
    out = torch.zeros(N, C_out, H_out, W_out)
    for i in range(H_out):
        for j in range(W_out):
            patch = x[:, :, i*stride:i*stride+kH, j*stride:j*stride+kW]
            # patch: (N, C_in, kH, kW)
            out[:, :, i, j] = (patch.unsqueeze(1) * w.unsqueeze(0)).sum(dim=(2,3,4))
    return out
```

### Max Pooling via Unfold

```python
def max_pool2d(x, k=2, stride=2):
    N, C, H, W = x.shape
    H_out = (H - k) // stride + 1
    W_out = (W - k) // stride + 1
    x_col = F.unfold(x, kernel_size=k, stride=stride)  # (N, C*k*k, H_out*W_out)
    x_col = x_col.view(N, C, k*k, H_out*W_out)
    out   = x_col.max(dim=2).values                    # (N, C, H_out*W_out)
    return out.view(N, C, H_out, W_out)
```

## W5 — 图神经网络（Week6 Tutorial）

### 构建归一化邻接矩阵

```python
def build_A_hat_norm(A):
    # A: (N, N) numpy, 无权重无向图
    n = A.shape[0]
    A_hat     = A + np.eye(n)                         # Ã = A + I（加自环）
    d_hat     = A_hat.sum(axis=1)                     # 每行的度
    D_inv_sqrt = np.diag(d_hat ** (-0.5))             # D̃^{-1/2}
    A_norm    = D_inv_sqrt @ A_hat @ D_inv_sqrt        # D̃^{-1/2} Ã D̃^{-1/2}
    return torch.tensor(A_norm, dtype=torch.float32)
```

### GCN Layer（Forward）

```python
class GCNLayer(nn.Module):
    def __init__(self, in_features, out_features, activation=True):
        super().__init__()
        self.W          = nn.Parameter(torch.empty(in_features, out_features))
        self.activation = activation
        nn.init.xavier_uniform_(self.W)

    def forward(self, H, A_hat_norm):
        agg = A_hat_norm @ H       # 邻域聚合：(N, F_in)
        out = agg @ self.W         # 线性变换：(N, F_out)
        if self.activation:
            out = F.relu(out)
        return out
```

### 两层 GCN（Node Classification）

```python
class GCN(nn.Module):
    def __init__(self, in_features, hidden, num_classes):
        super().__init__()
        self.gcn1 = GCNLayer(in_features, hidden,      activation=True)
        self.gcn2 = GCNLayer(hidden,      num_classes, activation=False)

    def forward(self, H, A_hat_norm):
        H = self.gcn1(H, A_hat_norm)   # (N, hidden)
        H = self.gcn2(H, A_hat_norm)   # (N, num_classes) — raw logits
        return H
```

### Kipf 公式总结

```
H^(l+1) = σ( D̃^{-1/2} Ã D̃^{-1/2} H^(l) W^(l) )

其中：
  Ã = A + I      （加自环）
  D̃ = degree matrix of Ã
  W^(l)          （可学习权重）
  σ              （ReLU，最后一层可省略）
```

## 速查：形状备忘

| 操作 | 输入 shape | 输出 shape |
|------|-----------|-----------|
| Linear `X@W` | `(N, F_in)` | `(N, F_out)` |
| Conv unfold | `(N, C, H, W)` | `(N, C·kH·kW, L)` where `L=H_out·W_out` |
| GCN agg `A@H` | `(N,N)·(N,F)` | `(N, F)` |
| BN `z_hat` | `(N, F)` | `(N, F)` （均值/方差按 dim=0） |
| Dropout mask | `(N, F)` | `(N, F)` （同 shape） |
