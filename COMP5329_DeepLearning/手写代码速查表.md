### Activation Functions（Forward + Backward）

```python
# Sigmoid
def forward(self, z):
    self.s = 1 / (1 + torch.exp(-z))
    return self.s

def backward(self, grad):
    return grad * self.s * (1 - self.s)

# ──────────────────────────────────────
# Tanh
def forward(self, z):
    self.out = torch.tanh(z)
    return self.out

def backward(self, grad):
    return grad * (1 - self.out ** 2)

# ──────────────────────────────────────
# ReLU
def forward(self, z):
    self.z = z
    return torch.clamp(z, min=0)

def backward(self, grad):
    return grad * (self.z > 0).float()
```

### HiddenLayer（Forward + Backward）

```python
def forward(self, X):
    self.X = X                           # (N, F_in)
    z = X @ self.W + self.b              # (N, F_out)
    return self.act.forward(z)           # activation applied

def backward(self, grad):
    dz = self.act.backward(grad)         # (N, F_out)
    self.dW = self.X.T @ dz             # (F_in, F_out)
    self.db = dz.sum(dim=0, keepdim=True)  # (1, F_out)
    dx = dz @ self.W.T                  # (N, F_in)
    return dx
```

### CrossEntropyLoss（Forward + Backward）

```python
def forward(self, logits, labels):
    # logits: (N, C), labels: (N,) int
    N = logits.shape[0]
    shifted = logits - logits.max(dim=1, keepdim=True).values  # numerical stability
    log_sum = torch.log(torch.exp(shifted).sum(dim=1, keepdim=True))
    log_probs = shifted - log_sum
    self.probs  = torch.exp(log_probs)  # save softmax probs
    self.labels = labels
    return -log_probs[torch.arange(N), labels].mean()

def backward(self):
    N = self.probs.shape[0]
    grad = self.probs.clone()            # (N, C)
    grad[torch.arange(N), self.labels] -= 1.0
    return grad / N
```

### Batch Normalization（Forward + Backward）

```python
# Forward
def forward(self, z):
    B = z.shape[0]
    mu    = z.mean(dim=0, keepdim=True)
    var   = z.var(dim=0, keepdim=True, unbiased=False)
    std   = torch.sqrt(var + 1e-5)
    z_hat = (z - mu) / std
    self._cache = (z_hat, std, B)
    return self.gamma * z_hat + self.beta

# Backward
def backward(self, grad):
    z_hat, std, B = self._cache
    self.dgamma = (grad * z_hat).sum(dim=0, keepdim=True)
    self.dbeta  =  grad.sum(dim=0, keepdim=True)
    dz_hat = grad * self.gamma
    dz = (1.0 / (B * std)) * (
        B * dz_hat
        - dz_hat.sum(dim=0, keepdim=True)
        - z_hat * (dz_hat * z_hat).sum(dim=0, keepdim=True)
    )
    return dz
```

### naive_conv2d（单样本，单通道理解版）

```python
def naive_conv2d(x, w, stride=1, padding=0):
    # x: (N, C_in, H, W)   w: (C_out, C_in, kH, kW)
    N, C_in, H, W = x.shape
    C_out, _, kH, kW = w.shape
    H_out = (H + 2*padding - kH) // stride + 1
    W_out = (W + 2*padding - kW) // stride + 1
    if padding > 0:
        x = F.pad(x, [padding]*4)
    out = torch.zeros(N, C_out, H_out, W_out)
    for i in range(H_out):
        for j in range(W_out):
            patch = x[:, :, i*stride:i*stride+kH, j*stride:j*stride+kW]
            # patch: (N, C_in, kH, kW)
            out[:, :, i, j] = (patch.unsqueeze(1) * w.unsqueeze(0)).sum(dim=(2,3,4))
    return out
```

### Max Pooling via Unfold

```python
def max_pool2d(x, k=2, stride=2):
    N, C, H, W = x.shape
    H_out = (H - k) // stride + 1
    W_out = (W - k) // stride + 1
    x_col = F.unfold(x, kernel_size=k, stride=stride)  # (N, C*k*k, H_out*W_out)
    x_col = x_col.view(N, C, k*k, H_out*W_out)
    out   = x_col.max(dim=2).values                    # (N, C, H_out*W_out)
    return out.view(N, C, H_out, W_out)
```

## W5 — 图神经网络（Week6 Tutorial）

### 构建归一化邻接矩阵

```python
def build_A_hat_norm(A):
    # A: (N, N) numpy, 无权重无向图
    n = A.shape[0]
    A_hat     = A + np.eye(n)                         # Ã = A + I（加自环）
    d_hat     = A_hat.sum(axis=1)                     # 每行的度
    D_inv_sqrt = np.diag(d_hat ** (-0.5))             # D̃^{-1/2}
    A_norm    = D_inv_sqrt @ A_hat @ D_inv_sqrt        # D̃^{-1/2} Ã D̃^{-1/2}
    return torch.tensor(A_norm, dtype=torch.float32)
```

### GCN Layer（Forward）

```python
class GCNLayer(nn.Module):
    def __init__(self, in_features, out_features, activation=True):
        super().__init__()
        self.W          = nn.Parameter(torch.empty(in_features, out_features))
        self.activation = activation
        nn.init.xavier_uniform_(self.W)

    def forward(self, H, A_hat_norm):
        agg = A_hat_norm @ H       # 邻域聚合：(N, F_in)
        out = agg @ self.W         # 线性变换：(N, F_out)
        if self.activation:
            out = F.relu(out)
        return out
```

### 两层 GCN（Node Classification）

```python
class GCN(nn.Module):
    def __init__(self, in_features, hidden, num_classes):
        super().__init__()
        self.gcn1 = GCNLayer(in_features, hidden,      activation=True)
        self.gcn2 = GCNLayer(hidden,      num_classes, activation=False)

    def forward(self, H, A_hat_norm):
        H = self.gcn1(H, A_hat_norm)   # (N, hidden)
        H = self.gcn2(H, A_hat_norm)   # (N, num_classes) — raw logits
        return H
```

### Kipf 公式总结

```
H^(l+1) = σ( D̃^{-1/2} Ã D̃^{-1/2} H^(l) W^(l) )

其中：
  Ã = A + I      （加自环）
  D̃ = degree matrix of Ã
  W^(l)          （可学习权重）
  σ              （ReLU，最后一层可省略）
```

## W7 — LSTM（Week7 Tutorial）

### LSTM 单步前向

```python
def lstm_forward(x_t, h_prev, c_prev, W_f, W_i, W_o, W_c):
    # 输入拼接：[h_{t-1}, x_t]
    combined = torch.cat([h_prev, x_t], dim=-1)

    # 三个门：全都是 sigmoid
    f_t = torch.sigmoid(combined @ W_f)   # 遗忘门：决定忘掉多少 c_{t-1}
    i_t = torch.sigmoid(combined @ W_i)   # 输入门：决定写入多少新信息
    o_t = torch.sigmoid(combined @ W_o)   # 输出门：决定读出多少 c_t

    # 候选细胞状态（用 tanh，范围 -1 到 1）
    c_tilde = torch.tanh(combined @ W_c)

    # 细胞状态更新：加法结构（不是乘法链！）
    c_t = c_prev * f_t + c_tilde * i_t

    # 输出
    h_t = o_t * torch.tanh(c_t)

    return h_t, c_t
```

**记忆要点**：三门 sigmoid，候选 tanh；细胞状态是**加法**更新（防梯度消失的关键）。

## W8 — Transformer & ViT & SSM（Week8 Tutorial）

### Scaled Dot-Product Attention

```python
def scaled_dot_product_attention(Q, K, V, mask=None):
    # Q: (B, N, d_k), K: (B, N, d_k), V: (B, N, d_v)
    d_k = Q.shape[-1]

    scores = Q @ K.transpose(-2, -1) / math.sqrt(d_k)  # (B, N, N)

    # 因果 mask（GPT 用，BERT 不用）
    if mask is not None:
        scores = scores.masked_fill(mask == 0, float('-inf'))

    weights = torch.softmax(scores, dim=-1)   # (B, N, N)
    output  = weights @ V                     # (B, N, d_v)
    return output
```

**记忆要点**：`Q @ K.T` → `(N, N)` 相似度矩阵；mask 用 `-inf` 填充后 softmax 为 0；最后乘 V 加权求和。

### ViT Patch Embedding

```python
class ViTPatchEmbedding(nn.Module):
    def __init__(self, img_size, patch_size, in_chans, embed_dim):
        super().__init__()
        self.num_patches = (img_size // patch_size) ** 2
        self.unfold    = nn.Unfold(kernel_size=patch_size, stride=patch_size)
        self.proj      = nn.Linear(in_chans * patch_size**2, embed_dim)
        self.cls_token = nn.Parameter(torch.zeros(1, 1, embed_dim))
        self.pos_embed = nn.Parameter(torch.zeros(1, 1 + self.num_patches, embed_dim))

    def forward(self, x):
        B = x.shape[0]
        patches = self.unfold(x)                    # (B, C*P*P, T)
        tokens  = self.proj(patches.transpose(1,2)) # (B, T, D)
        cls     = self.cls_token.expand(B, -1, -1)
        tokens  = torch.cat([cls, tokens], dim=1)   # (B, T+1, D)
        tokens  = tokens + self.pos_embed
        return tokens
```

### SSM 卷积核构建

```python
def build_ssm_kernel(A_bar, B_bar, C, L):
    K = torch.zeros(L)
    A_pow_B = B_bar.clone()          # A^0 B = B

    for i in range(L):
        K[i] = (C @ A_pow_B).squeeze()   # K_i = C * A_bar^i * B_bar
        A_pow_B = A_bar @ A_pow_B         # 递推

    return K   # y = conv(x, K)
```

**记忆要点**：用递推而不是直接算 $\bar{A}^i$，避免重复矩阵乘法。

## W12 — 生成模型（Week12 Tutorial）

### GAN 训练循环

```python
def train_gan_step(real_data, G, D, g_optim, d_optim):
    B = real_data.shape[0]

    # ── 训练 Discriminator ──
    z = torch.randn(B, latent_dim)
    fake_data = G(z).detach()              # .detach() 阻止梯度传回 G

    d_loss = (-torch.log(D(real_data) + 1e-8)
              - torch.log(1 - D(fake_data) + 1e-8)).mean()

    d_optim.zero_grad(); d_loss.backward(); d_optim.step()

    # ── 训练 Generator ──
    z = torch.randn(B, latent_dim)
    g_loss = -torch.log(D(G(z)) + 1e-8).mean()  # G 希望 D(G(z)) → 1

    g_optim.zero_grad(); g_loss.backward(); g_optim.step()
```

**记忆要点**：训练 D 时用 `.detach()` 截断梯度；G 的 loss = $-\log D(G(z))$（非饱和形式）。

### VAE 前向 + ELBO Loss

```python
class VAE(nn.Module):
    def encode(self, x):
        h = self.encoder(x)
        return self.fc_mu(h), self.fc_var(h)   # mu, log_var

    def reparameterize(self, mu, log_var):
        std = torch.exp(0.5 * log_var)
        eps = torch.randn_like(std)            # eps ~ N(0, I)
        return mu + std * eps                  # z ~ N(mu, std^2)

    def forward(self, x):
        mu, log_var = self.encode(x)
        z    = self.reparameterize(mu, log_var)
        x_hat = self.decode(z)

        recon_loss = F.mse_loss(x_hat, x, reduction='sum')
        kl_loss    = -0.5 * torch.sum(1 + log_var - mu**2 - torch.exp(log_var))
        return recon_loss + kl_loss
```

**记忆要点**：KL 解析解 $-\frac{1}{2}\sum(1+\log\sigma^2-\mu^2-\sigma^2)$；重参数化让梯度绕过采样流回 $\mu,\sigma$。

### DDPM 前向加噪 + 训练目标

```python
def q_sample(x0, t, alphas_bar):
    # Diffusion Kernel：x_t = sqrt(ᾱ_t) x0 + sqrt(1-ᾱ_t) ε
    sqrt_ab   = alphas_bar[t].sqrt().view(-1, 1, 1, 1)
    sqrt_1_ab = (1 - alphas_bar[t]).sqrt().view(-1, 1, 1, 1)
    eps = torch.randn_like(x0)
    return sqrt_ab * x0 + sqrt_1_ab * eps, eps

def ddpm_loss(model, x0, alphas_bar, T):
    t   = torch.randint(0, T, (x0.shape[0],))
    x_t, eps = q_sample(x0, t, alphas_bar)
    return F.mse_loss(model(x_t, t), eps)    # 预测加入的噪声
```

## 速查：形状备忘

| 操作 | 输入 shape | 输出 shape |
|------|-----------|-----------|
| Linear `X@W` | `(N, F_in)` | `(N, F_out)` |
| Conv unfold | `(N, C, H, W)` | `(N, C·kH·kW, L)` where `L=H_out·W_out` |
| GCN agg `A@H` | `(N,N)·(N,F)` | `(N, F)` |
| BN `z_hat` | `(N, F)` | `(N, F)` （均值/方差按 dim=0） |
| Dropout mask | `(N, F)` | `(N, F)` （同 shape） |
